```python
import random

# 状态定义
AP_STANDBY = 0
AP_TRANSITION = 1
AP_NOMINAL = 2
AP_MANEUVER = 3
AP_PULLUP = 4

S_NOMINAL = 0
S_FAULT = 1
S_TRANSITION = 2

def sensor_good(sensor_state):
    return sensor_state != S_FAULT

def request(ap_state):
    return ap_state in [AP_NOMINAL, AP_MANEUVER, AP_PULLUP]

def mode(ap_state):
    return ap_state == AP_NOMINAL

def next_ap_state(ap_state, sensor_state, standby, apfail, supported):
    if ap_state == AP_TRANSITION:
        if standby:
            return AP_STANDBY
        elif supported and sensor_good(sensor_state):
            return AP_NOMINAL
        else:
            return AP_TRANSITION
    elif ap_state == AP_NOMINAL:
        if standby:
            return AP_STANDBY
        elif not sensor_good(sensor_state):
            return AP_MANEUVER
        else:
            return AP_NOMINAL
    elif ap_state == AP_MANEUVER:
        if standby and sensor_good(sensor_state):
            return AP_STANDBY
        else:
            return AP_MANEUVER
    elif ap_state == AP_PULLUP:
        if supported and sensor_good(sensor_state):
            return AP_TRANSITION
        else:
            return AP_PULLUP
    elif ap_state == AP_STANDBY:
        if apfail:
            return AP_MANEUVER
        elif not standby:
            return AP_TRANSITION
        else:
            return AP_STANDBY
    else:
        return ap_state

def next_sensor_state(sensor_state, ap_state, limits):
    if sensor_state == S_NOMINAL:
        if limits:
            return S_FAULT
        elif not request(ap_state):
            return S_TRANSITION
        else:
            return S_NOMINAL
    elif sensor_state == S_FAULT:
        if not request(ap_state) and not limits:
            return S_TRANSITION
        else:
            return S_FAULT
    elif sensor_state == S_TRANSITION:
        if request(ap_state) and mode(ap_state):
            return S_NOMINAL
        else:
            return S_TRANSITION
    else:
        return sensor_state

def simulate(standby_seq, apfail_seq, supported_seq, limits_seq, initial_ap_state=AP_STANDBY, initial_sensor_state=S_NOMINAL):
    ap_state = initial_ap_state
    sensor_state = initial_sensor_state
    pullup_seq = []
    for t in range(len(standby_seq)):
        pullup = (ap_state == AP_MANEUVER)
        pullup_seq.append(pullup)
        # 当前输入
        standby = standby_seq[t]
        apfail = apfail_seq[t]
        supported = supported_seq[t]
        limits = limits_seq[t]
        # 计算下一个状态
        next_ap = next_ap_state(ap_state, sensor_state, standby, apfail, supported)
        next_sensor = next_sensor_state(sensor_state, ap_state, limits)
        ap_state = next_ap
        sensor_state = next_sensor
    return pullup_seq

def generate_random_sequence(length=11):
    standby_seq = [random.randint(0,1) for _ in range(length)]
    apfail_seq = [random.randint(0,1) for _ in range(length)]
    supported_seq = [random.randint(0,1) for _ in range(length)]
    limits_seq = [random.randint(0,1) for _ in range(length)]
    return standby_seq, apfail_seq, supported_seq, limits_seq
